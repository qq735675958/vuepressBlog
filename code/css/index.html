    # 常用的 CSS 片段

    # 清除浮动

    //清除浮动
.clearfix:after{    
    content: "\00A0";    
    display: block;    
    visibility: hidden;           
    width: 0;    
    height: 0;    
    clear: both;    
    font-size: 0;    
    line-height: 0;    
    overflow: hidden;
}
.clearfix{
    zoom: 1;
}

    # 垂直水平居中

    # 绝对定位方式且已知宽高

    margin值是宽高的一半

    position: absolute;
top: 50%;
left: 50%;
margin-top: -3em;
margin-left: -7em;
width: 14em;
height: 6em;

    # 绝对定位＋未知宽高＋ translate

    position: absolute;
left: 50%;
top: 50%;
transform: translate(-50%,-50%);

    # flex 轻松搞定水平垂直居中(未知宽高)

    display: flex;
align-items: center;
justify-content: center;
